\documentclass[border=5pt]{standalone}
\usepackage{tikz}
\usetikzlibrary{calc}
\usepackage{amsmath}
\begin{document}
    \begin{tikzpicture}[scale=2]
        % Draw unit circle
        \draw[blue] (0,0) circle (1);

        % Draw axes
        \draw[->] (-1.5,0) -- (1.5,0) node[right] {$x$};
        \draw[->] (0,-1.5) -- (0,1.5) node[above] {$y$};

        % Draw point (1,0)
        \fill (1,0) circle (1pt) node[below right] {$(1,0)$};

        % Draw point (sqrt(1-x^2),x) for x=0.6
        \def\x{0.6}
        \def\y{0.8}
        \fill (\y,\x) circle (1pt) node[right] {$(\sqrt{1-x^2},x)$};

        % Draw arc in red
        \draw[red,thick] (1,0) arc (0:37:1);

        % Draw angle theta
        \draw[red] (0.3,0) arc (0:37:0.3) node[midway,right] {$\theta$};

        % Draw height x
        \draw[dashed] (\y,\x) -- (\y,0);
        \draw[dashed] (\y,\x) -- (0,\x) node[left] {$x$};
        \node[below] at (0.6,0) {$\sqrt{1-x^2}$};

        \draw[green!50!black] (0,0) -- (\y,\x);

        % Label arc as arcsin(x)
        \node[red,right] at (1.2,0.3) {$\arcsin(x)=\theta$};
    \end{tikzpicture}
\end{document}
