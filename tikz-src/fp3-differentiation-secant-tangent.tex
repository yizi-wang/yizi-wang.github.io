\documentclass[border=5pt]{standalone}
\usepackage{tikz}
\usetikzlibrary{calc}
\usepackage{amsmath}
\begin{document}
    \begin{tikzpicture}[scale=1]
        % Draw curve
        \draw[domain=-2:2,smooth,variable=\x,blue] plot ({\x},{0.5*\x*\x});

        % Draw axes
        \draw[->] (-2.5,0) -- (2.5,0) node[right] {$x$};
        \draw[->] (0,-0.5) -- (0,3) node[above] {$y$};

        % Draw tangent line at x=1
        \draw[red,thick] (-1,-1.5) -- (3,2.5) node[right] {$y=f'(a)(x-a)+f(a)$};

        % Point of tangency
        \fill (1,0.5) circle (2pt);

        \node[right] at (1,0.5) {$P(a,f(a))$};
    \end{tikzpicture}
\end{document}
