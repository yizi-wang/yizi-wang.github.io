\documentclass[border=5pt]{standalone}
\usepackage{tikz}
\usetikzlibrary{calc}
\usepackage{amsmath}
\begin{document}
\begin{tikzpicture}[scale=1.2]
    % Axes
    \draw[->] (-0.5,0) -- (4,0) node[right] {$x$};
    \draw[->] (0,-0.5) -- (0,2) node[above] {$y$};

    % Plot 1/x
    \draw[thick, blue] plot[domain=1:4,samples=100] (\x,{1/\x}) node[right] {$y = \frac{1}{x}$};

    % Rotation indicators
    \draw[red,->] (2,0) arc (0:180:0.2);
    \draw[red] (2,0) arc (0:-180:0.2);

    % Labels
    \node at (1,-0.2) {$1$};
    \draw[dashed] (1,0) -- (1,1);
\end{tikzpicture}
