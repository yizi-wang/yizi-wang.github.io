\documentclass[border=5pt]{standalone}
\usepackage{tikz}
\usetikzlibrary{calc}
\usepackage{amsmath}
\begin{document}
    \begin{tikzpicture}[scale=1.2]
        % Draw the hyperbola
        \draw[blue, thick, domain=1.2:2, samples=50]
            plot (\x, {sqrt((\x*\x/1.44) - 1)});
        \draw[blue, thick, domain=1.2:2, samples=50]
            plot (\x, {-sqrt((\x*\x/1.44) - 1)});

        \draw[blue, thick, domain=-2:-1.2, samples=50]
            plot (\x, {sqrt((\x*\x/1.44) - 1)});
        \draw[blue, thick, domain=-2:-1.2, samples=50]
            plot (\x, {-sqrt((\x*\x/1.44) - 1)});

        \node[circle, fill, inner sep=1pt] (P) at (1.3,0.416) {};
            \node[right] at (P) {$P(a\cosh t,b\sinh t)$ or $P(a\sec \theta,b\tan \theta)$};
        % Draw asymptotes
        \draw[dashed] (-2,-1.66) -- (2,1.66) node[right] {$y = \frac{b}{a}x$};
        \draw[dashed] (-2,1.66) -- (2,-1.66) node[right] {$y = -\frac{b}{a}x$};

        % Draw axes
        \draw[->] (-2.5,0) -- (2.5,0) node[right] {$x$};
        \draw[->] (0,-2.5) -- (0,2.5) node[above] {$y$};

        % Label points
        \node[below right] at (1.2,0) {$a$};
        \node[below left] at (-1.2,0) {$-a$};
    \end{tikzpicture}
\end{document}
